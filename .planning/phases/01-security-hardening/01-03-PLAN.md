**Step 3 — Create idempotent migration script:**

Create `apps/api/src/lib/migrate-fallback-key.ts`:

```typescript
/**
 * One-time migration: encrypt any plaintext aiFallbackApiKey values in all tenant databases.
 * Safe to run multiple times (idempotent).
 *
 * Usage: pnpm tsx apps/api/src/lib/migrate-fallback-key.ts
 */
import { PrismaClient } from "@prisma/client";
import { encrypt } from "./crypto.js";
import { loadConfig } from "../config.js";

const isAlreadyEncrypted = (value: string): boolean => {
  const parts = value.split('.');
  if (parts.length !== 3) return false;
  try {
    return Buffer.from(parts[0], 'base64').length === 12;
  } catch {
    return false;
  }
};

export async function migrateFallbackApiKeys(
  prisma: PrismaClient,
  encryptionKey: string,
  schema: string = 'public'
): Promise<{ updated: number; skipped: number; errors: number }> {
  const results = { updated: 0, skipped: 0, errors: 0 };

  // Query chatbot_config rows with non-null aiFallbackApiKey
  const rows = await prisma.$queryRawUnsafe<Array<{ id: string; aiFallbackApiKey: string }>>(
    `SELECT id, "aiFallbackApiKey" FROM ${schema}."ChatbotConfig" WHERE "aiFallbackApiKey" IS NOT NULL`
  );

  for (const row of rows) {
    try {
      if (isAlreadyEncrypted(row.aiFallbackApiKey)) {
        results.skipped++;
        continue;
      }
      const encrypted = encrypt(row.aiFallbackApiKey, encryptionKey);
      await prisma.$executeRawUnsafe(
        `UPDATE ${schema}."ChatbotConfig" SET "aiFallbackApiKey" = $1 WHERE id = $2`,
        encrypted,
        row.id
      );
      results.updated++;
    } catch (err) {
      console.error(`[migrate-fallback-key] Error for row ${row.id}:`, err);
      results.errors++;
    }
  }

  return results;
}

// CLI entry point
if (process.argv[1] === new URL(import.meta.url).pathname) {
  const config = loadConfig();
  const prisma = new PrismaClient({ datasources: { db: { url: config.DATABASE_URL } } });
  migrateFallbackApiKeys(prisma, config.API_ENCRYPTION_KEY)
    .then((r) => {
      console.log(`[migrate-fallback-key] Done. Updated: ${r.updated}, Skipped: ${r.skipped}, Errors: ${r.errors}`);
      process.exit(r.errors > 0 ? 1 : 0);
    })
    .catch((err) => {
      console.error("[migrate-fallback-key] Fatal error:", err);
      process.exit(1);
    })
    .finally(() => prisma.$disconnect());
}
```

Note: The migration targets the default schema. For multi-tenant databases with per-tenant schemas, the caller must invoke `migrateFallbackApiKeys(prisma, key, tenantSchema)` for each tenant. Add a comment noting this.

**Anti-patterns:**
- Do NOT run the migration inside `loadConfig()` or application startup — it is a one-time deployment step, not a boot-time operation.
- Do NOT log the plaintext key values during migration — log only row IDs.
- Do NOT use `prisma.chatbotConfig.findMany()` in the migration if tenant schemas differ from the default Prisma schema — use raw SQL with the correct schema prefix.
