```typescript
import type pino from "pino";
// Import PrismaClient type for platformPrisma — adjust import path based on project structure
// e.g.: import type { PrismaClient } from "@prisma/client";

/**
 * Versioned migration record.
 * version format: "YYYY-MM-DD-NNN-description" (sorts lexicographically in apply order)
 */
interface Migration {
  version: string;
  description: string;
  sql: (schema: string) => string; // returns parameterized SQL string with schema interpolated
}

/**
 * All tenant schema migrations — ordered by version ascending.
 * Add new migrations HERE when new columns are needed.
 * NEVER remove or modify an existing migration version.
 *
 * Convention: version = ISO date + 3-digit sequence + short slug
 * Example: "2026-04-11-001-contact-tags"
 */
export const MIGRATIONS: Migration[] = [
  // Populate from the ALTER TABLE statements found in tenant-schema.ts.
  // Each ALTER TABLE becomes one entry:
  {
    version: "2026-04-11-001-contact-tags",
    description: "Add tags column to conversation table",
    sql: (schema) =>
      `ALTER TABLE ${schema}."Conversation" ADD COLUMN IF NOT EXISTS "tags" TEXT[] NOT NULL DEFAULT '{}';`
  },
  // ... one entry per ALTER TABLE statement found in tenant-schema.ts
  // Executor: read tenant-schema.ts and create one Migration per ALTER TABLE
];

/**
 * Run all pending migrations for a single tenant.
 * Per locked decision D-MIGRATION-FAIL:
 *   - Errors are caught per-tenant
 *   - Returns "failed" without throwing
 *   - Caller handles startup summary logging
 */
export async function runMigrations(
  platformPrisma: { $executeRawUnsafe: (sql: string) => Promise<unknown>; $queryRawUnsafe: <T>(sql: string) => Promise<T> },
  tenantId: string,
  logger: pino.Logger
): Promise<"success" | "skipped" | "failed"> {
  // Resolve tenant schema name — match existing pattern in database.ts
  // e.g.: const schema = `"tenant_${tenantId.replace(/-/g, '_')}"`;
  // Read database.ts to find the exact schema naming function and use it.
  const schema = resolveTenantSchema(tenantId); // implement per database.ts pattern

  try {
    // Step 1: Ensure migrations table exists
    await platformPrisma.$executeRawUnsafe(
      `CREATE TABLE IF NOT EXISTS ${schema}."schema_migrations" (` +
      `"version" TEXT PRIMARY KEY, ` +
      `"appliedAt" TIMESTAMPTZ NOT NULL DEFAULT NOW()` +
      `);`
    );

    // Step 2: Fetch applied versions
    const applied = await platformPrisma.$queryRawUnsafe<{ version: string }[]>(
      `SELECT "version" FROM ${schema}."schema_migrations";`
    );
    const appliedSet = new Set(applied.map((r) => r.version));

    // Step 3: Check if all migrations already applied
    const pending = MIGRATIONS.filter(m => !appliedSet.has(m.version));
    if (pending.length === 0) return "skipped";

    // Step 4: Apply pending migrations in order
    for (const migration of pending) {
      try {
        await platformPrisma.$executeRawUnsafe(migration.sql(schema));
        await platformPrisma.$executeRawUnsafe(
          `INSERT INTO ${schema}."schema_migrations" ("version") VALUES ('${migration.version}');`
        );
      } catch (err) {
        // Per D-MIGRATION-FAIL: log structured error, return "failed"
        logger.error(
          { tenantId, migration: migration.version, error: err },
          "Migration failed — tenant will operate on partial schema"
        );
        return "failed";
      }
    }

    return "success";
  } catch (err) {
    // Outer catch: migrations table creation failed or schema doesn't exist
    logger.error(
      { tenantId, migration: "schema_migrations-create", error: err },
      "Migration infrastructure failed for tenant"
    );
    return "failed";
  }
}

/** Resolve tenant schema name — must match the naming logic in database.ts */
function resolveTenantSchema(tenantId: string): string {
  // Read database.ts to find the exact implementation.
  // Example (adjust to match actual code):
  return `"tenant_${tenantId.replace(/-/g, "_")}"`;
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04-01 | Injection | MIGRATIONS[].sql via $executeRawUnsafe | mitigate | Migration SQL is static strings defined in source code — not user-supplied input. Schema name is interpolated via `resolveTenantSchema()` which must produce only alphanumeric+underscore values. Validate schema name format: `if (!/^"tenant_[a-z0-9_]+"$/.test(schema)) throw new Error("Invalid schema name")` |
| T-02-04-02 | Information Disclosure | Tenant IDs in Pino startup summary log | accept | Startup logs are server-side only (not API responses); tenant IDs are internal identifiers, not secrets |
| T-02-04-03 | Denial of Service | Migration loop blocks startup for many tenants | mitigate | Each runMigrations() call is lightweight (1–2 SQL queries if skipped). Cap acceptable via current platform scale. Add 5-second timeout per tenant if needed in future |
| T-02-04-04 | Tampering | MIGRATIONS[] version collision | mitigate | Version strings are PRIMARY KEY in schema_migrations — duplicate version insert raises unique constraint error, caught per-migration and logged. Document: version strings must be unique across all history |
| T-02-04-05 | Elevation of Privilege | Migration SQL running as platform DB user | accept | Platform DB user already has schema-level DDL rights (used by ensureSchema today); no privilege escalation |
