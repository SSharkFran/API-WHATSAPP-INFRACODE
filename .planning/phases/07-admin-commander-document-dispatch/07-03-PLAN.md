```typescript
import { randomUUID } from 'node:crypto';
import type pino from 'pino';

export type AdminActionType =
  | 'document_send'
  | 'session_close'
  | 'status_query'
  | 'metrics_query'
  | 'human_takeover';

export interface AdminActionLogEntry {
  id?: string;
  triggeredByJid: string;
  actionType: AdminActionType;
  targetContactJid?: string | null;
  documentName?: string | null;
  messageText?: string | null;
  deliveryStatus?: 'pending' | 'sent' | 'failed';
}

export interface AdminActionLogDeps {
  logger: pino.Logger;
  getTenantDb: (tenantId: string) => {
    $executeRawUnsafe: (sql: string, ...args: unknown[]) => Promise<unknown>;
  };
  getTenantSchema: (tenantId: string) => string; // e.g. "tenant_abc123"
}

export class AdminActionLogService {
  constructor(private readonly deps: AdminActionLogDeps) {}

  /**
   * Non-blocking write via setImmediate — never throws.
   * Errors are logged as warnings.
   */
  write(tenantId: string, entry: AdminActionLogEntry): void {
    setImmediate(() => {
      void this.insertRow(tenantId, entry).catch((err) =>
        this.deps.logger.warn({ err, tenantId, entry }, '[AdminActionLogService] write failed')
      );
    });
  }

  private async insertRow(tenantId: string, entry: AdminActionLogEntry): Promise<void> {
    const db = this.deps.getTenantDb(tenantId);
    const schema = this.deps.getTenantSchema(tenantId);
    const id = entry.id ?? randomUUID();
    await db.$executeRawUnsafe(
      `INSERT INTO "${schema}"."AdminActionLog"
         ("id", "triggeredByJid", "actionType", "targetContactJid", "documentName", "messageText", "deliveryStatus", "createdAt")
       VALUES ($1, $2, $3, $4, $5, $6, $7, NOW())`,
      id,
      entry.triggeredByJid,
      entry.actionType,
      entry.targetContactJid ?? null,
      entry.documentName ?? null,
      entry.messageText ?? null,
      entry.deliveryStatus ?? 'pending'
    );
  }
}
```

Key constraints:
- `write()` returns `void` — caller does NOT await it
- `setImmediate` defers the async insert — message pipeline is never blocked
- `$executeRawUnsafe` with positional `$1`–`$7` params — NO string interpolation
- Schema comes from `getTenantSchema()` — never computed inline

```typescript
export interface AdminActionLogEntry {
  id: string;
  triggeredByJid: string;
  actionType: string;
  targetContactJid: string | null;
  documentName: string | null;
  messageText: string | null;
  deliveryStatus: string;
  createdAt: string; // ISO timestamp
}

export async function getTenantActionHistory(limit = 100): Promise<AdminActionLogEntry[]> {
  try {
    const res = await fetch(`${API_BASE}/tenant/action-history?limit=${limit}`, {
      cache: 'no-store',
    });
    if (!res.ok) return [];
    return (await res.json()) as AdminActionLogEntry[];
  } catch {
    return [];
  }
}
```

Also add a backend API route: `apps/api/src/modules/tenants/routes.ts` (or wherever tenant routes live) — add:
```typescript
// GET /tenant/action-history?limit=N
// Returns AdminActionLog rows for the requesting tenant, ordered by createdAt DESC
fastify.get('/tenant/action-history', {
  schema: { querystring: { limit: { type: 'integer', default: 100, maximum: 500 } } },
  preHandler: [fastify.authenticate],
}, async (request, reply) => {
  const { tenantId } = request.auth;
  const { limit } = request.query as { limit: number };
  const db = tenantPrismaRegistry.get(tenantId);
  const schema = getTenantSchema(tenantId);
  const rows = await db.$queryRawUnsafe<AdminActionLogEntry[]>(
    `SELECT id, "triggeredByJid", "actionType", "targetContactJid", "documentName", "messageText", "deliveryStatus", "createdAt"
     FROM "${schema}"."AdminActionLog"
     ORDER BY "createdAt" DESC
     LIMIT $1`,
    limit
  );
  return reply.send(rows);
});
```
